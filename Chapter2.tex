\begin{equation} 
\nabla \cdot E = 0,  \\ 
\nabla \times = - d_{t} B, \\
\end{equation}
\begin{equation}
\nabla \cdot B = 0, \\
\nabla \times B=  \frac{_{t} E}{v^2},  \\
\end{equation}
\begin{equation}
\hat{E_{0}} = E_{x}\hat{x} + E_{y}\hat{y}+E_{z}\hat{z}
\end{equation}
\begin{equation}
\hat{B_{0}} = B_{x}\hat{x} + B_{y}\hat{y}+B_{z}\hat{z}
\end{equation}
\begin{equation}
d_{y}E_{x}-d_{x}E_{y}= i w B_{z}
\end{equation}
\begin{equation}
i k E_{x}-d_{x}E_{z}= i w B_{y}
\end{equation}
\begin{equation}
d_{y}E_{z}-i k E_{y}= i w B_{x}
\end{equation}
\begin{equation}
d_{x}B_{z}-d_{y}B_{x}= \frac{-iw E_{Z} }{v^2} 
\end{equation}
\begin{equation}
i k B_{x}-d_{x}B_{z}= \frac{-iw E_{y} }{v^2} 
\end{equation}
\begin{equation}
d_{y}B_{z}- i k B_{y}= \frac{-iw E_{x} }{v^2} 
\end{equation}
